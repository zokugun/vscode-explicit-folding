Manually controls how and where to fold your code

## Configuration

In your Settings:

```
"folding": {
    "*": {
        "begin": "{{{",
        "end": "}}}"
    },
    "typescript": {
        "begin": "#region",
        "end": "#endregion"
    },
    "javascriptreact": [
        {
            "begin": "{/*",
            "end": "*/}"
        },
        {
            "begin": "<",
            "end": "/>"
        }
    ],
    "php":[
        {
            "comment":"//"
        },
        {
            "begin": "\/*",
            "end": "*\/",
            "offsetTop":1,
        },
        {
            "begin":"<?",
            "end":"?>"
        },
        {
            "begin": "{",
            "end": "}"
        },
        {
            "begin": "[",
            "end": "]"
        },
        {
            "begin": "(",
            "end": ")"
        },
    ]
}
```

or with **regex**:

```
"folding": {
    "*": {
        "beginRegex": "\\{\\{\\{",
        "endRegex": "\\}\\}\\}"
    },
    "typescript": {
        "beginRegex": "#region",
        "endRegex": "#endregion"
    },
    "javascriptreact": [
        {
            "beginRegex": "\\{/\\*",
            "endRegex": "\\*/\\}"
        },
        {
            "begin": "<",
            "end": "/>"
        }
    ],
    "php":{
        "comment":"//",
        "beginRegex": "\\(|\\[|\\{|\\<\\?",
        "endRegex": "\\}|\\]|\\)|\\?\\>"
    }
}
```

## Migrate from v0.2 to v0.3

The configuration's properties `begin` and `end` should be renamed as `beginRegex` and `endRegex`.
